% ------------------------------------------------------------
\begin{frame}{攻擊技術說明 T1548.002}
\begin{itemize}
  \item \textbf{技術名稱}：Abuse Elevation Control Mechanism - Bypass User Account Control
  \item \textbf{子技術}：Slui File Handler Hijack
  \item \textbf{原理}：
  \begin{itemize}
    \item 利用 \texttt{slui.exe} 在啟動時會讀取 \texttt{HKCU\textbackslash Software\textbackslash Classes\textbackslash exefile} File Handler 的特性
    \item 由於 \texttt{slui.exe} 具有 \texttt{autoElevate=true} 屬性，以 High Integrity 執行
    \item 透過 Hijack Registry，使其以高權限執行任意程式
    \item 從 Medium Integrity $\rightarrow$ High Integrity，\textbf{不需使用者互動}
  \end{itemize}
  \item \textbf{工具}：UACME (\texttt{Akagi64.exe}) Method 45
\end{itemize}
\end{frame}

\begin{frame}[fragile]{Procmon 輔助分析 - Registry Hijack}
Procmon 補充 Sysmon 所看不到的 Registry 操作，揭露 Method 45 核心手法：
\begin{lstlisting}
01:09:44.952  Akagi64.exe  RegCreateKey
   HKCU\Software\Classes\exefile\shell\open\command
   Disposition: REG_OPENED_EXISTING_KEY

01:09:44.953  Akagi64.exe  RegSetValue
   HKCU\Software\Classes\exefile\shell\open\command\(Default)
   Type: REG_SZ   Data: C:\Windows\System32\cmd.exe

01:09:44.958  Akagi64.exe  QueryDirectory
   C:\Windows\System32\slui.exe
\end{lstlisting}

\textbf{手法說明}：攻擊者將 \texttt{exefile} 的 shell open command 改寫為 \texttt{cmd.exe}，
接著觸發具有 \texttt{autoElevate} 屬性的 \texttt{slui.exe}，
由其以 High Integrity 啟動被劫持的 \texttt{cmd.exe}。
\end{frame}

\begin{itemize}

\end{itemize}
